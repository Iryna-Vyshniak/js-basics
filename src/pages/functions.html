1 & 2. Способи створення функцій та підрахунок аргументів



// 1. Function Declaration (Оголошення функції)
function calculateSum(a, b) {
  return a + b;
}

// 2. Function Expression (Функціональний вираз)
const multiply = function(a, b) {
  return a * b;
};

// 3. Arrow Function (Стрілкова функція)
const divide = (a, b) => a / b;

// 4. IIFE (Негайно викликана функція)
(function() {
  console.log("Викликається відразу");
})();

// 5. Async Function (Асинхронна функція)
async function fetchData() {
  return await fetch('/api/data');
}

// 6. Generator Function (Функція-генератор)
function* idMaker() {
  let index = 0;
  while (true) yield index++;
}

// 7. Object / Class Methods (Методи об'єктів та класів)
const apiService = {
  getData() { return true; }
};

// 8. The Function Constructor (Не рекомендовано)
const dynamicSum = new Function('a', 'b', 'return a + b');


// === ПІДРАХУНОК АРГУМЕНТІВ ===

const countRest = (...args) => args.length;

// 2. Псевдомасив arguments - Застарілий підхід (відсутній у стрілкових)
function countLegacy() {
  return arguments.length;
}
